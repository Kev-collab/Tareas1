do {

    // Generar un número aleatorio entre 1 y 9 para la computadora

    let numeroComputadora = Math.floor(Math.random() * 9) + 1;

    let numeroUsuario;

    

    // Validar que el usuario ingrese un número entre 3 y 6

    do {

        numeroUsuario = parseInt(prompt("Ingrese un número entre 3 y 6:"));

    } while (numeroUsuario < 3 || numeroUsuario > 6);

    

    // Pedir al usuario que adivine si su número es mayor, menor o igual

    let eleccion = prompt("¿Cree que su número es mayor, menor o igual al número de la computadora? (mayor/menor/igual)");

    

    let resultado;

    if (numeroUsuario > numeroComputadora) {

        resultado = "mayor";

    } else if (numeroUsuario < numeroComputadora) {

        resultado = "menor";

    } else {

        resultado = "igual";

    }

    

    // Mostrar los resultados

    let mensaje = `La computadora eligió ${numeroComputadora}, usted eligió ${numeroUsuario}. Su elección es ${resultado}. `;

    mensaje += (eleccion === resultado) ? "¡Ha adivinado!" : "No ha adivinado.";

    alert(mensaje);

    

    // Preguntar si desea jugar otra vez

    var jugarDeNuevo = prompt("¿Quiere jugar otra vez? (SI/NO)").toUpperCase();

} while (jugarDeNuevo === "SI");
